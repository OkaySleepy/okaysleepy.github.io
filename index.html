Event Listener

@EventHandler
public void onPlayerJoin(PlayerJoinEvent event) {
    Player player = event.getPlayer();
    player.sendMessage(ChatColor.GREEN + "Welcome to the server!");
    player.playSound(player.getLocation(), Sound.ENTITY_PLAYER_LEVELUP, 1.0f, 1.0f);
    // Add custom join effects
    player.addPotionEffect(new PotionEffect(
        PotionEffectType.GLOWING, 200, 1, true, true
    ));
}

Custom NBT Handling

public ItemStack createCustomItem(Material material, String displayName, String nbtKey, String nbtValue) {
    ItemStack item = new ItemStack(material);
    ItemMeta meta = item.getItemMeta();
    meta.setDisplayName(displayName);
    item.setItemMeta(meta);
    
    // Add NBT data
    NBTItem nbtItem = new NBTItem(item);
    nbtItem.setString(nbtKey, nbtValue);
    return nbtItem.getItem();
}

GUI Creation

public Inventory createShopGUI() {
    Inventory inv = Bukkit.createInventory(null, 27, "Server Shop");
    
    // Add items to the shop
    ItemStack sword = new ItemStack(Material.DIAMOND_SWORD);
    ItemMeta meta = sword.getItemMeta();
    meta.setDisplayName(ChatColor.AQUA + "Diamond Sword");
    meta.setLore(Arrays.asList(ChatColor.GRAY + "Price: 10 diamonds"));
    sword.setItemMeta(meta);
    
    // Add to inventory
    inv.setItem(12, sword);
    return inv;
}
